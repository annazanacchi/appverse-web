24  package org.appverse.web.framework.backend.api.helpers.security;
25  
26  import java.util.ArrayList;
27  import java.util.Collection;
28  import java.util.HashSet;
29  import java.util.List;
30  
31  import javax.servlet.http.HttpServletRequest;
32  
33  import org.springframework.security.core.GrantedAuthority;
34  import org.springframework.security.core.authority.SimpleGrantedAuthority;
35  import org.springframework.security.core.authority.mapping.MappableAttributesRetriever;
36  import org.springframework.security.web.authentication.preauth.PreAuthenticatedGrantedAuthoritiesWebAuthenticationDetails;
37  import org.springframework.security.web.authentication.preauth.j2ee.J2eeBasedPreAuthenticatedWebAuthenticationDetailsSource;
38  
39  public class AK47AuthenticationUserDetailsSource extends
40  		J2eeBasedPreAuthenticatedWebAuthenticationDetailsSource {
41  
42  	private String authoritiesSessionAttribute = "GMEC_ENTITLEMENTS";
43  
44  	@Override
45  	@SuppressWarnings("unchecked")
46  	public PreAuthenticatedGrantedAuthoritiesWebAuthenticationDetails buildDetails(
47  			HttpServletRequest request) {
48  		if (request.getSession().getAttribute(authoritiesSessionAttribute) != null) {
49  			List<String> authorities = (List<String>) request.getSession()
50  					.getAttribute(authoritiesSessionAttribute);
51  			Collection<GrantedAuthority> grantedAuthorities = new ArrayList<GrantedAuthority>();
52  			for (String authority : authorities) {
53  				GrantedAuthority grantedAuthority = new SimpleGrantedAuthority(
54  						authority);
55  				grantedAuthorities.add(grantedAuthority);
56  			}
57  			PreAuthenticatedGrantedAuthoritiesWebAuthenticationDetails details = new PreAuthenticatedGrantedAuthoritiesWebAuthenticationDetails(
58  					request, grantedAuthorities);
59  			return details;
60  		} else {
61  			GrantedAuthority grantedAuthority = new SimpleGrantedAuthority(
62  					"NO_AUTHORITY");
63  			Collection<GrantedAuthority> grantedAuthorities = new ArrayList<GrantedAuthority>();
64  			grantedAuthorities.add(grantedAuthority);
65  			PreAuthenticatedGrantedAuthoritiesWebAuthenticationDetails details = new PreAuthenticatedGrantedAuthoritiesWebAuthenticationDetails(
66  					request, grantedAuthorities);
67  			return details;
68  		}
69  	}
70  
71  	public void setAuthoritiesSessionAttribute(
72  			String authoritiesSessionAttribute) {
73  		this.authoritiesSessionAttribute = authoritiesSessionAttribute;
74  	}
75  
76  	@Override
77  	public void setMappableRolesRetriever(
78  			MappableAttributesRetriever aJ2eeMappableRolesRetriever) {
79  		this.j2eeMappableRoles = new HashSet<String>();
80  	}
81  
82  }
