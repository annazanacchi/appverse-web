24  package org.appverse.web.framework.backend.api.converters.helpers;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  import org.appverse.web.framework.backend.api.common.AbstractBean;
30  import org.appverse.web.framework.backend.api.converters.IBeanConverter;
31  
32  /**
33   * Wraps a ArrayList class addind properties to manage detach and undetach
34   * operations needed to perform lazy loading operations to ArrayList collections
35   * 
36   * @see org.appverse.web.framework.backend.api.converters.helpers.ConverterUtils
37   * @see org.appverse.web.framework.backend.aop.aspects.ConverterAspect
38   * 
39   */
40  public class IndirectArrayList<E, F> extends ArrayList<E> {
41  
42  	private static final long serialVersionUID = -1841312699228737735L;
43  	private boolean detached = true;
44  	private List<F> source = null;
45  	private Class<E> itemType = null;
46  	private IBeanConverter itemConverter = null;
47  
48  	/**
49  	 * Creates a new LazyArrayList with a source Collection and a itemType and
50  	 * itemConverter. The ConverterAspect will use the itemType and
51  	 * itemConverter to convert the source collection to actual collection
52  	 * 
53  	 * @param source
54  	 *            Source collection to be converted
55  	 * @param itemType
56  	 *            Type of items of actual collection
57  	 * @param itemConverter
58  	 *            Coverter to be used to convert items from source collection to
59  	 *            actual collection
60  	 */
61  	public IndirectArrayList(List<F> source, Class<E> itemType,
62  			IBeanConverter itemConverter) {
63  		super();
64  		detached = true;
65  		this.source = source;
66  		this.itemType = itemType;
67  		this.itemConverter = itemConverter;
68  	}
69  
70  	/**
71  	 * Returns the element at the specified position in this list.
72  	 * ConverterAspect need to LazyArrayList override this method
73  	 * 
74  	 * @param index
75  	 *            index of the element to return
76  	 * @return the element at the specified position in this list
77  	 * @throws IndexOutOfBoundsException
78  	 *             {@inheritDoc}
79  	 */
80  	@Override
81  	public E get(int index) {
82  		return super.get(index);
83  	}
84  
85  	public IBeanConverter getItemConverter() {
86  		return itemConverter;
87  	}
88  
89  	public Class<?> getItemType() {
90  		return itemType;
91  	}
92  
93  	@SuppressWarnings("unchecked")
94  	public List<AbstractBean> getSource() {
95  		return (List<AbstractBean>) source;
96  	}
97  
98  	/**
99  	 * Returns if collection is a proxy that need be converted (detached true)
100 	 * or conversion has been performed and list has the actual items (detached
101 	 * false)
102 	 * 
103 	 * @return
104 	 */
105 	public boolean isDetached() {
106 		return detached;
107 	}
108 
109 	/**
110 	 * Clean the elements used to helps lazy loading. This method is called when
111 	 * the conversion has been performed
112 	 * 
113 	 */
114 	public void undetach() {
115 		detached = false;
116 		source = null;
117 		itemType = null;
118 		itemConverter = null;
119 	}
120 }
