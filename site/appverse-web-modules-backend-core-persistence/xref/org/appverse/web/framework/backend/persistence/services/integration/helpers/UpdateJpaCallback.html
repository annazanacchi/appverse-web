24  package org.appverse.web.framework.backend.persistence.services.integration.helpers;

26  import java.util.ArrayList;
27  import java.util.List;
28  import java.util.Map;
29  
30  import javax.persistence.EntityManager;
31  import javax.persistence.PersistenceException;
32  import javax.persistence.Query;
33  
34  public class UpdateJpaCallback<T> {
35  
36  	private Query query;
37  	private String queryString;
38  	private List<QueryJpaCallbackParameter> namedParameters;
39  	private Object[] indexedParameters;
40  	private List<QueryJpaCallbackHint> hints;
41  
42  	public UpdateJpaCallback(String queryString) {
43  		super();
44  		if (queryString==null || queryString.isEmpty()){
45  			throw new PersistenceException("query cannot be null or empty");
46  		}
47  		this.queryString = queryString;
48  	}
49  	
50  	public UpdateJpaCallback(Query query) {
51  		super();
52  		if(query==null){
53  			throw new PersistenceException("query cannot be null");
54  		}
55  		this.query = query;
56  	}	
57  
58  	public Object doInJpa(EntityManager em) throws PersistenceException {
59  
60  		// create query if it was set as a queryString
61  		if(query==null){
62  			query = em.createQuery(queryString);
63  		}		
64  
65  		// set parameters
66  		if (namedParameters != null) {
67  			for (QueryJpaCallbackParameter namedParameter : namedParameters) {
68  				query.setParameter(namedParameter.getName(),
69  						namedParameter.getValue());
70  			}
71  		} else if (indexedParameters != null) {
72  			for (int i = 0; i < indexedParameters.length; i++) {
73  				query.setParameter(i + 1, indexedParameters[i]);
74  			}
75  		}
76  
77  		// set query hints
78  		if (hints != null) {
79  			for (QueryJpaCallbackHint hint : hints) {
80  				query.setHint(hint.getHint(), hint.getValue());
81  			}
82  		}
83  
84  		return query.executeUpdate();
85  	}
86  
87  	public void setHints(List<QueryJpaCallbackHint> hints) {
88  		this.hints = hints;
89  	}
90  
91  	public void setIndexedParameters(Object[] indexedParameters) {
92  		this.indexedParameters = indexedParameters;
93  	}
94  
95  	public void setNamedParameters(List<QueryJpaCallbackParameter> parameters) {
96  		this.namedParameters = parameters;
97  	}
98  
99  	public void setNamedParameters(Map<String, Object> parameters) {
100 		List<QueryJpaCallbackParameter> queryJpaCallbackParameters = new ArrayList<QueryJpaCallbackParameter>();
101 		for (String key : parameters.keySet()) {
102 			QueryJpaCallbackParameter queryJpaCallbackParameter = new QueryJpaCallbackParameter(
103 					key, parameters.get(key));
104 			queryJpaCallbackParameters.add(queryJpaCallbackParameter);
105 		}
106 		this.namedParameters = queryJpaCallbackParameters;
107 	}
108 }
