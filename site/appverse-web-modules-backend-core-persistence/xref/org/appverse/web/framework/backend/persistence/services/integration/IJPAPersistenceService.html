38  /**
39   * This is the interface for JPA Persistence services implementations providing
40   * support for JPA operations.
41   * 
42   * @param <T>
43   *            The integration bean type managed by your integration service
44   */
45  public interface IJPAPersistenceService<T extends AbstractIntegrationBean> {
46  
47  	/**
48  	 * Performs JPA contains operation
49  	 * 
50  	 * @param beanP
51  	 * @throws Exception
52  	 */
53  	void contains(final T beanP) throws Exception;
54  
55  	/**
56  	 * This methods count the number of T elements
57  	 * 
58  	 * @return int The number of records found
59  	 */
60  	public int count() throws Exception;
61  
62  	/**
63  	 * This method allows you to execute a count passing an
64  	 * IntegrationDataFilter as a parameter. You can pass an
65  	 * IntegrationPaginatedDataFilter as a parameter in order to perform counts
66  	 * related to remote pagination. The IntegrationDataFilter allows you to
67  	 * apply very simple conditions over <T> beans in order to filter the
68  	 * results to be retrieved.
69  	 * 
70  	 * @param filter
71  	 * @return int The number of records found
72  	 * @throws Exception
73  	 */
74  	int count(final IntegrationDataFilter filter) throws Exception;

76  	/**
77  	 * This methods allows you to perform any JPA operation providing direct
78  	 * access to the EntityManager by means a QueryJpaCallback<T> object for <T>
79  	 * beans for a count operation. If you want predefined standard hints, set
80  	 * QueryJPACallback "standardHints" boolean parameter to true.
81  	 * 
82  	 * @param query
83  	 * @return
84  	 * @throws Exception
85  	 */
86  	int count(QueryJpaCallback<T> query) throws Exception;

89  	 * This methods count the number of T elements returned by a JPQL query

91  	 * @param queryString

94  	 * @throws Exception
95  	 */
96  	int count(String queryString) throws Exception;

124 	/**
125 	 * This methods allows you to perform JPA operation providing direct access
126 	 * to the EntityManager by means a QueryJpaCallback<Object[]> returning the
127 	 * number of objects found. Use this method when you need to return
128 	 * directlyobjects that does not match neither the type of the managed
129 	 * objects of your repository <T> a result of your query nor a "resultType".
130 	 * A typical usage is when you do not want to retrieve a full object or a
131 	 * combination of different attributes of different objects involved in the
132 	 * query or just calculated fields. In summary, use this method when you do
133 	 * not have an IntegrationBean or "resultType" to act as a container of the
134 	 * data returned by the query. In this case you get directly a List with an
135 	 * array of objects including the selected data. Of course, in this case,
136 	 * you are responsible to parse this data to objects so do not use this
137 	 * method if you have an specific object to hold the data.
138 	 * 
139 	 * @param query
140 	 * @return int The number of records found
141 	 * @throws Exception
142 	 */
143 	int countObjectArray(QueryJpaCallback<Object[]> query) throws Exception;
