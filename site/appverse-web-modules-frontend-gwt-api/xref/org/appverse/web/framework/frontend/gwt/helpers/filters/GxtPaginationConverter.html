33  import com.sencha.gxt.data.shared.SortDir;
34  import com.sencha.gxt.data.shared.SortInfo;
35  import com.sencha.gxt.data.shared.loader.PagingLoadConfig;
36  import com.sencha.gxt.data.shared.loader.PagingLoadResult;
37  import com.sencha.gxt.data.shared.loader.PagingLoadResultBean;
38  
39  public class GxtPaginationConverter {
40  
41  	@SuppressWarnings({ "unchecked", "rawtypes" })
42  	public static PagingLoadResult<? extends AbstractPresentationBean> convert(
43  			final GWTPresentationPaginatedResult<? extends AbstractPresentationBean> paginatedResult) {
44  		return new PagingLoadResultBean(paginatedResult.getData(),
45  				paginatedResult.getTotalLength(), paginatedResult.getOffset());
46  	}
47  
48  	public static GWTPresentationPaginatedDataFilter convert(
49  			final PagingLoadConfig pagingLoadConfig) {
50  		GWTPresentationPaginatedDataFilter filter = new GWTPresentationPaginatedDataFilter();
51  		filter.setLimit(pagingLoadConfig.getLimit());
52  		filter.setOffset(pagingLoadConfig.getOffset());
53  		List<? extends SortInfo> sortInfoList = pagingLoadConfig.getSortInfo();
54  		for (SortInfo sortInfo : sortInfoList) {
55  			// filter.getColumnsToSort().add(sortInfo.getSortField());
56  			String sortingDirection = "";
57  			if (sortInfo.getSortDir() == SortDir.ASC) {
58  				// filter.getSortingDirections().add(
59  				// PresentationPaginatedDataFilter.ASC);
60  				sortingDirection = PresentationDataFilter.ASC;
61  			} else {
62  				// filter.getSortingDirections().add(
63  				// PresentationPaginatedDataFilter.DESC);
64  				sortingDirection = PresentationDataFilter.DESC;
65  			}
66  			filter.addSortingColumn(sortInfo.getSortField(), sortingDirection);
67  		}
68  		return filter;
69  	}
