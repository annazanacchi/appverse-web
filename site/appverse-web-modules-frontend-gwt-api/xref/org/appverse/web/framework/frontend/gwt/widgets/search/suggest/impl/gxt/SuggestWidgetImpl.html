167 	public GWTPresentationPaginatedDataFilter getDataFilter() {
168 		dataFilter.resetConditions();
169 
170 		if ((searchText.getText().trim() != null)
171 				&& (searchText.getText().trim().length() > 0)) {
172 			// dataFilter.getValues().add(
173 			// PresentationDataFilter.WILDCARD_ALL + searchText.getText()
174 			// + PresentationDataFilter.WILDCARD_ALL);
175 			// dataFilter.getColumns().add(getModelSearchField());
176 			// dataFilter.getLikes().add(true);
177 
178 			if (isIgnoreCase()) {
179 				dataFilter.addLikeConditionIgnoreCase(
180 						getModelSearchField(),
181 						PresentationDataFilter.WILDCARD_ALL
182 								+ searchText.getText()
183 								+ PresentationDataFilter.WILDCARD_ALL);
184 			} else {
185 				dataFilter.addLikeCondition(
186 						getModelSearchField(),
187 						PresentationDataFilter.WILDCARD_ALL
188 								+ searchText.getText()
189 								+ PresentationDataFilter.WILDCARD_ALL);
190 			}
191 		}
192 
193 		return dataFilter;
194 	}

208 	private void initLoaders() {
209 		/**** SUGGEST *****/
210 
211 		// proxy for getting suggestions from server
212 		final RpcProxy<PagingLoadConfig, PagingLoadResult<M>> proxySuggest = new RpcProxy<PagingLoadConfig, PagingLoadResult<M>>() {
213 			@Override
214 			public void load(final PagingLoadConfig loadConfig,
215 					final AsyncCallback<PagingLoadResult<M>> callback) {
216 				GWT.log("Proxy load method called");
217 				// We clean any possible previously selected value
218 				value = null;
219 				callbackSuggest = new AsyncCallback<GWTPresentationPaginatedResult<M>>() {
220 					@Override
221 					public void onFailure(final Throwable caught) {
222 						GWT.log("list load failed!! " + caught.getMessage());
223 					}
224 
225 					@Override
226 					@SuppressWarnings("unchecked")
227 					public void onSuccess(
228 							final GWTPresentationPaginatedResult<M> result) {
229 						GWT.log("list loaded!! ");
230 
231 						final PagingLoadResult<M> loadResult = (PagingLoadResult<M>) GxtPaginationConverter
232 								.convert(result);
233 						callback.onSuccess(loadResult);
234 
235 					}
236 				};
237 				loadConfig.setSortInfo(new ArrayList<SortInfo>());
238 				dataFilter = GxtPaginationConverter.convert(loadConfig);
239 
240 				// CAUTION
241 				// You must use "getDataFilter()" to avoid loosing any kind of
242 				// filter when using toolbar
243 
244 				handlerManager.fireEvent(new LoadSuggestEvent<M>(
245 						getDataFilter(), callbackSuggest));
246 
247 			}
248 		};
249 
250 		final PagingLoader<PagingLoadConfig, PagingLoadResult<M>> listLoaderSuggest = new PagingLoader<PagingLoadConfig, PagingLoadResult<M>>(
251 				proxySuggest);
252 
253 		final LoadResultListStoreBinding<PagingLoadConfig, M, PagingLoadResult<M>> handlerSuggest = new LoadResultListStoreBinding<PagingLoadConfig, M, PagingLoadResult<M>>(
254 				listStoreSuggest);
255 
256 		listLoaderSuggest.addLoadHandler(handlerSuggest);
257 		searchText.setLoader(listLoaderSuggest);
258 
259 		/* Combo selection clicked */
260 		searchText.addBeforeSelectionHandler(new BeforeSelectionHandler<M>() {
261 			@Override
262 			public void onBeforeSelection(final BeforeSelectionEvent<M> event) {
263 				event.cancel();
264 				M model = searchText.getListView().getSelectionModel()
265 						.getSelectedItem();
266 				setValue(model);
267 				handlerManager.fireEvent(new SelectSuggestEvent<M>(model));
268 			}
269 		});
270 
271 		searchText.addBlurHandler(new BlurHandler() {
272 			@Override
273 			public void onBlur(final BlurEvent event) {
274 				// If no value has been selected and force selection flag is
275 				// true, we clear the field
276 				if (isForceSelection() && value == null) {
277 					clear();
278 				}
279 			}
280 		});
281 
282 		/* set to null selected value, since criteria has changed */
283 		//workaround to solve keypressed event bug. workaround is not documented.
284 		//http://code.google.com/p/google-web-toolkit/issues/detail?id=72
285 		searchText.addKeyUpHandler(new KeyUpHandler() {
286 			@Override
287 			public void onKeyUp(final KeyUpEvent event) {
288 				if (!oldText.equals(searchText.getText()))
289 					setValue(null);
290 			}
291 
292 		});
293 
294 		/**************************************************************/
295 		/* setPageSize MUST be called AFTER setting loader in combobox. */
296 		/**************************************************************/
297 		searchText.setPageSize(10);
298 
299 	}

302 	 * This method will only create the widgets that are marked as provided. It
303 	 * does not add the widget to the panel: this is still done by the UI
304 	 * binder.
305 	 */
306 	private void initProvidedWidgets() {
307 
308 		// list store
309 		listStoreSuggest = new ListStore<M>(props.id());
310 
311 		// final Bundle b = GWT.create(Bundle.class);
312 		// b.css().ensureInjected();
313 
314 		listViewSuggest = new ListView<M, M>(listStoreSuggest,
315 				new IdentityValueProvider<M>());
316 
317 		listViewSuggest.setCell(new AbstractCell<M>() {
318 
319 			@Override
320 			public void render(
321 					final com.google.gwt.cell.client.Cell.Context context,
322 					final M value, final SafeHtmlBuilder sb) {
323 				appearance.render(sb, value, template);
324 			}
325 		});
326 		final LabelProvider<M> title = props.label();
327 
328 		final ComboBoxCell<M> cell = new ComboBoxCell<M>(listStoreSuggest,
329 				title, listViewSuggest) {
330 			@Override
331 			public M selectByValue(final String value) {
332 				M item = super.selectByValue(value);
333 				if (item == null) {
334 					item = instance;
335 					props.name().setValue(item, value);
336 				}
337 				return item;
338 			}
339 		};
340 
341 		searchText = new ComboBox<M>(cell);
342 		searchText.setHideTrigger(true);
343 		searchText.setTriggerAction(ComboBoxCell.TriggerAction.ALL);
344 		searchText.setForceSelection(true);
345 	}
