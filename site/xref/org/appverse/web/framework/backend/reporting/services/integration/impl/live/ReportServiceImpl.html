41  import org.appverse.web.framework.backend.api.services.business.AbstractBusinessService;
42  import org.appverse.web.framework.backend.reporting.services.integration.ReportService;
43  import org.springframework.stereotype.Service;

45  @Service("reportService")
46  public class ReportServiceImpl extends AbstractBusinessService implements
47  		ReportService {
48  
49  	@Override
50  	public byte[] generateExcel(List<ExcelSheetReportData> excelSheetsReportData)
51  			throws Exception {
52  
53  		if (excelSheetsReportData == null || excelSheetsReportData.size() == 0) {
54  			throw new Exception("There are no data to make report.");
55  		}
56  
57  		String[] sheetNamesArray = new String[excelSheetsReportData.size()];
58  		List<JasperPrint> jasperPrints = new ArrayList<JasperPrint>();
59  		int i = 0;
60  		for (ExcelSheetReportData excelSheetReportData : excelSheetsReportData) {
61  			sheetNamesArray[i] = excelSheetReportData.getSheetName();
62  			i++;
63  
64  			JRDataSource reportDataSource = new JRMapCollectionDataSource(
65  					excelSheetReportData.getSheetData());
66  			JasperPrint jasperPrint = null;
67  			if (excelSheetReportData.getSheetReportLocation() != null
68  					&& !excelSheetReportData.getSheetReportLocation()
69  							.equals("")) {
70  				jasperPrint = JasperFillManager.fillReport(
71  						excelSheetReportData.getSheetReportLocation(),
72  						excelSheetReportData.getSheetParameters(),
73  						reportDataSource);
74  			} else {
75  				jasperPrint = JasperFillManager.fillReport(
76  						excelSheetReportData.getSheetReport(),
77  						excelSheetReportData.getSheetParameters(),
78  						reportDataSource);
79  			}
80  			jasperPrints.add(jasperPrint);
81  		}
82  
83  		JasperPrint firstJasperPrint = jasperPrints.get(0);
84  		if (jasperPrints.size() > 1) {
85  			for (i = 1; i < jasperPrints.size(); i++) {
86  				List<JRPrintPage> additionalPages = new ArrayList<JRPrintPage>(
87  						jasperPrints.get(i).getPages());
88  				int fistJasperPrintPages = firstJasperPrint.getPages().size();
89  				for (int count = 0; count < additionalPages.size(); count++) {
90  					firstJasperPrint.addPage(fistJasperPrintPages,
91  							additionalPages.get(count));
92  				}
93  			}
94  		}
95  
96  		JRExporter exporter = new JExcelApiExporter();
97  		exporter.setParameter(JRXlsExporterParameter.JASPER_PRINT,
98  				firstJasperPrint);
99  		exporter.setParameter(JRXlsExporterParameter.IS_DETECT_CELL_TYPE,
100 				Boolean.TRUE);
101 		exporter.setParameter(JRXlsExporterParameter.IS_WHITE_PAGE_BACKGROUND,
102 				Boolean.FALSE);
103 
104 		exporter.setParameter(
105 				JRXlsExporterParameter.IS_REMOVE_EMPTY_SPACE_BETWEEN_ROWS,
106 				Boolean.TRUE);
107 		exporter.setParameter(
108 				JRXlsExporterParameter.IS_REMOVE_EMPTY_SPACE_BETWEEN_COLUMNS,
109 				Boolean.TRUE);
110 		exporter.setParameter(JRXlsExporterParameter.IS_ONE_PAGE_PER_SHEET,
111 				Boolean.TRUE);
112 		exporter.setParameter(JRXlsExporterParameter.SHEET_NAMES,
113 				sheetNamesArray);
114 		exporter.setParameter(JExcelApiExporterParameter.IS_COLLAPSE_ROW_SPAN,
115 				Boolean.TRUE);
116 		ByteArrayOutputStream outputStream = new ByteArrayOutputStream(32768);
117 		exporter.setParameter(JRXlsExporterParameter.OUTPUT_STREAM,
118 				outputStream);
119 		// exporter.setParameter(JRXlsExporterParameter.OUTPUT_FILE_NAME,
120 		// "C:/development/workspaces/jasper/report1.xls");
121 		exporter.exportReport();
122 		return outputStream.toByteArray();
123 	}
