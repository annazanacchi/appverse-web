24  package org.appverse.web.framework.backend.frontfacade.json.controllers;

26  import java.io.BufferedReader;
27  import java.io.ByteArrayOutputStream;
28  import java.io.IOException;
29  import java.io.InputStreamReader;
30  import java.nio.charset.Charset;
31  
32  import org.codehaus.jackson.JsonEncoding;
33  import org.codehaus.jackson.JsonGenerator;
34  import org.codehaus.jackson.map.ObjectMapper;
35  import org.codehaus.jackson.map.type.TypeFactory;
36  import org.codehaus.jackson.type.JavaType;
37  import org.slf4j.Logger;
38  import org.slf4j.LoggerFactory;
39  import org.springframework.http.HttpInputMessage;
40  import org.springframework.http.HttpOutputMessage;
41  import org.springframework.http.MediaType;
42  import org.springframework.http.converter.AbstractHttpMessageConverter;
43  import org.springframework.http.converter.HttpMessageNotReadableException;
44  import org.springframework.http.converter.HttpMessageNotWritableException;
45  import org.springframework.stereotype.Component;
46  import org.springframework.util.Assert;
47  
48  @Component
49  public class CustomMappingJacksonHttpMessageConverter extends
50  		AbstractHttpMessageConverter<Object> {
51  
52  	private final Logger logger = LoggerFactory
53  			.getLogger(CustomMappingJacksonHttpMessageConverter.class);
54  
55  	public static final Charset DEFAULT_CHARSET = Charset.forName("UTF-8");
56  
57  	private ObjectMapper objectMapper = new ObjectMapper();
58  
59  	private boolean prefixJson = false;
60  
61  	/**
62  	 * Construct a new {@code BindingJacksonHttpMessageConverter}.
63  	 */
64  	public CustomMappingJacksonHttpMessageConverter() {
65  		super(new MediaType("application", "json", DEFAULT_CHARSET));
66  	}
67  
68  	@Override
69  	public boolean canRead(Class<?> clazz, MediaType mediaType) {
70  		JavaType javaType = getJavaType(clazz);
71  		return this.objectMapper.canDeserialize(javaType) && canRead(mediaType);
72  	}
73  
74  	@Override
75  	public boolean canWrite(Class<?> clazz, MediaType mediaType) {
76  		return this.objectMapper.canSerialize(clazz) && canWrite(mediaType);
77  	}
78  
79  	@Override
80  	protected Long getContentLength(Object o, MediaType contentType) {
81  		try {
82  			ByteArrayOutputStream baos = new ByteArrayOutputStream();
83  			JsonGenerator baosjsonGenerator = this.objectMapper
84  					.getJsonFactory().createJsonGenerator(baos);
85  			this.objectMapper.writeValue(baosjsonGenerator, o);
86  			ByteArrayOutputStream os = (ByteArrayOutputStream) baosjsonGenerator
87  					.getOutputTarget();
88  			return new Long(os.size());
89  		} catch (Exception e) {
90  			return new Long(0);
91  		}
92  
93  	}
94  
95  	private JsonEncoding getEncoding(MediaType contentType) {
96  		if (contentType != null && contentType.getCharSet() != null) {
97  			Charset charset = contentType.getCharSet();
98  			for (JsonEncoding encoding : JsonEncoding.values()) {
99  				if (charset.name().equals(encoding.getJavaName())) {
100 					return encoding;
101 				}
102 			}
103 		}
104 		return JsonEncoding.UTF8;
105 	}

107 	/**
108 	 * Returns the Jackson {@link JavaType} for the specific class.
109 	 * 
110 	 * <p>
111 	 * Default implementation returns
112 	 * {@link TypeFactory#type(java.lang.reflect.Type)}, but this can be
113 	 * overridden in subclasses, to allow for custom generic collection
114 	 * handling. For instance:
115 	 * 
116 	 * <pre class="code">
117 	 * protected JavaType getJavaType(Class&lt;?&gt; clazz) {
118 	 * 	if (List.class.isAssignableFrom(clazz)) {
119 	 * 		return TypeFactory.collectionType(ArrayList.class, MyBean.class);
120 	 * 	} else {
121 	 * 		return super.getJavaType(clazz);
122 	 * 	}
123 	 * }
124 	 * </pre>
125 	 * 
126 	 * @param clazz
127 	 *            the class to return the java type for
128 	 * @return the java type
129 	 */
130 	protected JavaType getJavaType(Class<?> clazz) {
131 		return TypeFactory.defaultInstance().constructType(clazz);
132 	}
133 
134 	@Override
135 	protected Object readInternal(Class<?> clazz, HttpInputMessage inputMessage)
136 			throws IOException, HttpMessageNotReadableException {
137 		BufferedReader bufferedReader = new BufferedReader(
138 				new InputStreamReader(inputMessage.getBody(), DEFAULT_CHARSET));
139 		StringBuilder stringBuilder = new StringBuilder();
140 		String line = null;
141 
142 		while ((line = bufferedReader.readLine()) != null) {
143 			stringBuilder.append(line + " ");
144 		}
145 
146 		logger.debug("Middleware recieves " + clazz.getName() + " "
147 				+ stringBuilder.toString());
148 		JavaType javaType = getJavaType(clazz);
149 		return this.objectMapper.readValue(stringBuilder.toString(), javaType);
150 	}
151 
152 	protected Object readInternal(Class<?> clazz, String inputMessage)
153 			throws IOException, HttpMessageNotReadableException {
154 
155 		logger.debug("Middleware recieves " + clazz.getName() + " "
156 				+ inputMessage);
157 		JavaType javaType = getJavaType(clazz);
158 		return this.objectMapper.readValue(inputMessage, javaType);
159 	}

161 	/**
162 	 * Sets the {@code ObjectMapper} for this view. If not set, a default
163 	 * {@link ObjectMapper#ObjectMapper() ObjectMapper} is used.
164 	 * <p>
165 	 * Setting a custom-configured {@code ObjectMapper} is one way to take
166 	 * further control of the JSON serialization process. For example, an
167 	 * extended {@link org.codehaus.jackson.map.SerializerFactory} can be
168 	 * configured that provides custom serializers for specific types. The other
169 	 * option for refining the serialization process is to use Jackson's
170 	 * provided annotations on the types to be serialized, in which case a
171 	 * custom-configured ObjectMapper is unnecessary.
172 	 */
173 	public void setObjectMapper(ObjectMapper objectMapper) {
174 		Assert.notNull(objectMapper, "'objectMapper' must not be null");
175 		this.objectMapper = objectMapper;
176 	}
177 
178 	/**
179 	 * Indicates whether the JSON output by this view should be prefixed with
180 	 * "{} &&". Default is false.
181 	 * <p>
182 	 * Prefixing the JSON string in this manner is used to help prevent JSON
183 	 * Hijacking. The prefix renders the string syntactically invalid as a
184 	 * script so that it cannot be hijacked. This prefix does not affect the
185 	 * evaluation of JSON, but if JSON validation is performed on the string,
186 	 * the prefix would need to be ignored.
187 	 */
188 	public void setPrefixJson(boolean prefixJson) {
189 		this.prefixJson = prefixJson;
190 	}
191 
192 	@Override
193 	protected boolean supports(Class<?> clazz) {
194 		// should not be called, since we override canRead/Write instead
195 		throw new UnsupportedOperationException();
196 	}
197 
198 	@Override
199 	protected void writeInternal(Object o, HttpOutputMessage outputMessage)
200 			throws IOException, HttpMessageNotWritableException {
201 		JsonEncoding encoding = getEncoding(outputMessage.getHeaders()
202 				.getContentType());
203 		JsonGenerator jsonGenerator = this.objectMapper.getJsonFactory()
204 				.createJsonGenerator(outputMessage.getBody(), encoding);
205 		if (this.prefixJson) {
206 			jsonGenerator.writeRaw("{} && ");
207 		}
208 		this.objectMapper.writeValue(jsonGenerator, o);
209 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
210 		JsonGenerator baosjsonGenerator = this.objectMapper.getJsonFactory()
211 				.createJsonGenerator(baos, encoding);
212 		this.objectMapper.writeValue(baosjsonGenerator, o);
213 		logger.debug("Middleware returns " + o.getClass().getName() + " "
214 				+ baos.toString());
215 	}
