49  @Controller
50  @RequestMapping(value = "/*/services")
51  public class GwtRpcController extends RemoteServiceServlet {
52  
53  	private static final long serialVersionUID = 4147354200774086984L;
54  
55  	@Autowired
56  	private ApplicationContext applicationContext;
57  	@Autowired
58  	private ServletContext servletContext;
59  	private final ThreadLocal<String> serviceName = new ThreadLocal<String>();
60  
61  	private void checkXSRFToken(final HttpServletRequest request)
62  			throws IOException {
63  		String requestValue = request.getHeader("X-XSRF-Cookie");
64  		String sessionValue = (String) request.getSession().getAttribute(
65  				"X-XSRF-Cookie");
66  		if (sessionValue != null && !sessionValue.equals(requestValue)) {
67  			throw new PreAuthenticatedCredentialsNotFoundException(
68  					"XSRF attribute not found in session.");
69  
70  		}
71  	}
72  
73  	private String createXSRFToken(final HttpServletRequest request)
74  			throws IOException {
75  		HttpSession session = request.getSession();
76  		String xsrfSessionToken = (String) session
77  				.getAttribute("X-XSRF-Cookie");
78  		if (xsrfSessionToken == null) {
79  			Random r = new Random(System.currentTimeMillis());
80  			long value = System.currentTimeMillis() + r.nextLong();
81  			char ids[] = session.getId().toCharArray();
82  			for (int i = 0; i < ids.length; i++) {
83  				value += ids[i] * (i + 1);
84  			}
85  			xsrfSessionToken = Long.toString(value);
86  			session.setAttribute("X-XSRF-Cookie", xsrfSessionToken);
87  		}
88  		return xsrfSessionToken;
89  	}
90  
91  	@Override
92  	public ServletContext getServletContext() {
93  		return servletContext;
94  	}
95  
96  	@RequestMapping(value = "/*.rpc")
97  	public ModelAndView handleRequest(final HttpServletRequest request,
98  			final HttpServletResponse response) throws Exception {
99  		String path = request.getServletPath();
100 		serviceName.set(path.substring(path.lastIndexOf('/') + 1,
101 				path.lastIndexOf('.')));
102 		Object presentationService = applicationContext.getBean(serviceName
103 				.get());
104 		if (!(presentationService instanceof AuthenticationServiceFacade)) {
105 			checkXSRFToken(request);
106 		}
107 		super.doPost(request, response);
108 		return null;
109 	}
110 
111 	@Override
112 	public String processCall(final String payload)
113 			throws SerializationException {
114 		try {
115 			Object presentationService = applicationContext.getBean(serviceName
116 					.get());
117 			if (!(presentationService instanceof RemoteService)) {
118 				throw new IllegalArgumentException(
119 						"Requested Spring Bean is not a GWT RemoteService Presentation Service: "
120 								+ payload + " (" + presentationService + ")");
121 			}
122 
123 			RPCRequest rpcRequest = RPC.decodeRequest(payload,
124 					presentationService.getClass(), this);
125 			if (presentationService instanceof AuthenticationServiceFacade
126 					&& rpcRequest.getMethod().equals(
127 							AuthenticationServiceFacade.class
128 									.getMethod("getXSRFSessionToken"))) {
129 				return RPC.encodeResponseForSuccess(rpcRequest.getMethod(),
130 						createXSRFToken(getThreadLocalRequest()));
131 			}
132 			return RPC.invokeAndEncodeResponse(presentationService,
133 					rpcRequest.getMethod(), rpcRequest.getParameters(),
134 					rpcRequest.getSerializationPolicy(), rpcRequest.getFlags());
135 		} catch (Exception e) {
136 			GWTPresentationException pex = new GWTPresentationException(
137 					e.getMessage());
138 			return RPC.encodeResponseForFailure(null, pex);
139 		}
140 	}
