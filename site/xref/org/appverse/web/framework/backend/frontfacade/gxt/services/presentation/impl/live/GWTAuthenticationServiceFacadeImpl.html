47  	@Autowired
48  	private AuthenticationManager authenticationManager;

50  	/**
51  	 * Used by jUnit tests to create a security context to call operations and
52  	 * for a principal to be obtained by those operations.
53  	 */
54  	@Override
55  	public void authenticatePrincipal(final String principal,
56  			final List<String> credentials) {
57  
58  		List<GrantedAuthority> authList = new ArrayList<GrantedAuthority>();
59  		User user = new User(principal, "", true, true, true, true, authList);
60  		Authentication authentication = new PreAuthenticatedAuthenticationToken(
61  				user, credentials);
62  		authenticationManager.authenticate(authentication);
63  	}
64  
65  	/**
66  	 * Takes the username and password as provided and checks the validaty of
67  	 * the credentials. Spring security is used to check the credentielas and to
68  	 * return the authenticated principal with it's authorized roles. An
69  	 * exception is thrown if the authentication failes.
70  	 * 
71  	 * @param username
72  	 *            String containing the username of the principal to login
73  	 * @param password
74  	 *            String containing the password used to identify the current
75  	 *            user
76  	 * @return AuthorizationDataFO object containing the name of the principal
77  	 *         and the authorized roles.
78  	 */
79  	@SuppressWarnings("unchecked")
80  	@Override
81  	public GWTAuthorizationDataVO authenticatePrincipal(final String username,
82  			final String password) {
83  		final UsernamePasswordAuthenticationToken usernamePasswordAuthenticationToken = new UsernamePasswordAuthenticationToken(
84  				username, password);
85  		final Authentication authentication = authenticationManager
86  				.authenticate(usernamePasswordAuthenticationToken);
87  		SecurityContextHolder.getContext().setAuthentication(authentication);
88  		final Collection<GrantedAuthority> authorities = (Collection<GrantedAuthority>) SecurityContextHolder
89  				.getContext().getAuthentication().getAuthorities();
90  		final List<String> grantedRoles = new ArrayList<String>();
91  		for (final GrantedAuthority grantedAuthority : authorities) {
92  			grantedRoles.add(grantedAuthority.getAuthority());
93  		}
94  		final String name = SecurityContextHolder.getContext()
95  				.getAuthentication().getName();
96  		return new GWTAuthorizationDataVO(grantedRoles, name);
97  	}
98  
99  	@SuppressWarnings("unchecked")
100 	@Override
101 	public List<String> getAuthorities() {
102 		final Authentication authentication = SecurityContextHolder
103 				.getContext().getAuthentication();
104 		List<String> credentials = new ArrayList<String>();
105 		Collection<GrantedAuthority> grantedAuthorities = (Collection<GrantedAuthority>) authentication
106 				.getAuthorities();
107 		for (GrantedAuthority grantedAuthority : grantedAuthorities) {
108 			credentials.add(grantedAuthority.getAuthority());
109 		}
110 		return credentials;
111 	}
112 
113 	@Override
114 	public String getPrincipal() {
115 		final Authentication authentication = SecurityContextHolder
116 				.getContext().getAuthentication();
117 		return authentication.getName();
118 	}

135 	/**
136 	 * Checks if the current user is authenticated by evaluating the
137 	 * SecurityContext
138 	 * 
139 	 * @return Boolean true if the current user is authenticated and false
140 	 *         elsewise
141 	 */
142 	@Override
143 	public boolean isPrincipalAuthenticated() {
144 		final Authentication authentication = SecurityContextHolder
145 				.getContext().getAuthentication();
146 		if (authentication != null) {
147 			final User userPrincipal = (User) authentication.getPrincipal();
148 			return userPrincipal != null;
149 		} else {
150 			return false;
151 		}
