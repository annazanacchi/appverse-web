92  	public void defaultPresentationExceptionTreatmentWithoutClosingWindow(
93  			final GWTPresentationException ex) {
94  		NotificationManager.showError("Error Message:"
95  				+ ex.getMessage()
96  				+ "Error Time:"
97  				+ DateTimeFormat.getFormat("yyyy-MM-dd'T'HH:mm:ss.SSSZZZ")
98  						.format(new Date()));
99  	}

103 	 * default name in your application context. This is the default treatment for

128 	/**
129 	 * Default onFailure() method. If the exception is an instance of
130 	 * PresentationException and a code is informed we call
131 	 * handleApplicationExeption which provides a default treatment translating
132 	 * the exception code to a message and showing a dialog to the user without
133 	 * closing the browser tab. handleApplicationException can be overriden if
134 	 * we want to implement especific PresentationException treatment or just
135 	 * close the browser tab.
136 	 * If receive StatusCodeException, check expired session and call 
137 	 * handleExpiredSessionException(), which can be overridden
138 	 * 
139 	 * @param Throwable
140 	 *            Exception to handle.
141 	 */
142 	@Override
143 	public void onFailure(final Throwable ex) {
144 		GWTPresentationException pex = null;
145 		if (ex instanceof GWTPresentationException) {
146 			pex = (GWTPresentationException) ex;
147 		} else if (ex instanceof StatusCodeException) {
148 			if (((StatusCodeException) ex).getStatusCode() == CUSTOM_SESSION_EXPIRED_CODE)
149 			{
150 				handleExpiredSessionException();
151 				return;
152 			}
153 			else
154 				return;
155 			// Application exception (with an application code)
156 		} else {
157 			// We should not receive exceptions different from
158 			// GWTPresentationException in RPC commands. We control this just
159 			// for security.
160 			pex = new GWTPresentationException(ex.getMessage(), null);
161 		}
162 		handlePresentationException(pex);
163 		return;
164 	}
