30  import com.google.gwt.http.client.Request;

33  import com.google.gwt.http.client.RequestException;

38  /*
39   * LogoutManagementRpcCommandImpl integrates GWT logout with Spring Security.
40   * The request builder will post a request to the default Spring Security logout URL "/j_spring_security_logout".
41   * Spring Security mechanism will logout the user in the server. If this is done successfully, the GWT status it is 
42   * cleared reloading the page. This is perfect as ensures that all data in front end it is cleared and the front end application
43   * side is ready for another user to log-in.
44   * The proper Spring security will be in charge to show the log-in page after the user has been logged out.
45   * 
46   * Spring security setup example:
47   * 
48   * - Spring Security filter in web.xml:
49   * 
50   * 	<filter>
51   *		<filter-name>springSecurityFilterChain</filter-name>
52   *		<filter-class>org.springframework.web.filter.DelegatingFilterProxy</filter-class>
53   *	</filter>
54   * 
55   * 	<filter-mapping>
56   *		<filter-name>springSecurityFilterChain</filter-name>
57   *		<url-pattern>/*</url-pattern>
58   *	</filter-mapping>
59   * 
60   * - Spring security setup example:
61   *
62   *  <http auto-config='true'>
63   *    <!-- http://blog.idm.fr/2010/09/spring-security-redirecting-to-faviconico.html -->
64   *    <intercept-url pattern="/login.jsp*" access="IS_AUTHENTICATED_ANONYMOUSLY"/>
65   *    <intercept-url pattern="/favicon.ico" access="IS_AUTHENTICATED_ANONYMOUSLY" />
66   *    <intercept-url pattern="/**" access="ROLE_APP_ADMIN_CONSOLE_ACCESS" />
67   *    <form-login login-page='/login.jsp'/>
68   *    <logout logout-success-url="/login.jsp"/>    
69   *  </http>	   
70   */
71  @EventHandler
72  public class LogoutManagementRpcCommandImpl extends
73  		AbstractRpcCommand<FrameworkEventBus> implements
74  		LogoutManagementRpcCommand {
75  
76  	@Override
77  	public void onLogout() {
78  		RequestBuilder rb = new RequestBuilder(RequestBuilder.POST,
79  				"j_spring_security_logout");
80  		try {
81  			rb.sendRequest(null, new RequestCallback() {
82  				@Override
83  				public void onError(final Request request,
84  						final Throwable caught) {
85  					// Logout in server failed
86  				}
87  
88  				@Override
89  				public void onResponseReceived(final Request request,
90  						final Response response) {
91  					// Reloading the page will clear GWT status.
92  					// As the user has been logged out, spring security will
93  					// show login page again
94  					Window.Location.reload();
95  				}
96  			});
97  		} catch (RequestException re) {
98  
99  		}
100 	}
101 }
